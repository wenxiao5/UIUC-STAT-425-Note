## $B
##  [1] 14.0  8.8 11.2 14.2 11.8  6.4  9.8 11.3  9.3 13.6
# Create paired figure
matplot(cbind(shoes$A,shoes$B),pch=c("A","B"),xlab="Boy",ylab="Wear")

### Now we’ll do a one-sample t-test for the difference between
### the shoe with material A and the shoe with material B

t.test(shoes$A - shoes$B)
## 
##  One Sample t-test
## 
## data:  shoes$A - shoes$B
## t = -3.3489, df = 9, p-value = 0.008539
## alternative hypothesis: true mean is not equal to 0
## 95 percent confidence interval:
##  -0.6869539 -0.1330461
## sample estimates:
## mean of x 
##     -0.41
### Better yet, we find the p-value of the
### one sample t-statistic by using the
### randomization distribution.
### Here we must randomize within each pair

(shoes$A - shoes$B)*
  sample(c(-1,1),10,replace=TRUE)
##  [1]  0.8 -0.6  0.3 -0.1 -1.1  0.2  0.3  0.5  0.5 -0.3
tstats <- replicate(10000, 
                    t.test( (shoes$A - shoes$B)*
                            sample(c(-1,1),10,replace=TRUE) )$statistic)

t.observed <- t.test(shoes$A - shoes$B)$statistic

pval <- mean(abs(tstats) >= abs(t.observed))

pval
## [1] 0.0166
hist(tstats, breaks=50, freq=FALSE, col="grey90",
     main="Randomization Distrib. (Sim.), with t.observed and t(9) density")

abline(v=c(t.observed,-t.observed), lty=2, col="red")

lines(seq(-5,5,len=100), dt(seq(-5,5,len=100), df=9), col="blue")
